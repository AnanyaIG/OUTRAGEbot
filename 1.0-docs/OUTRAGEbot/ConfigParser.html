ConfigParser class for OUTRAGEbot

This class deals with the parsing of the bots.
This class loads the bots details from the INI file, then executes it. This enables the bot to have a very 	modular and easily modifiable structure. Due to the fact that this has moved from the rather redundant 	PHP based configuration, here is a sample (containing all the current values) of the new config file.
	; 
	; This header has a '~' suffix, which denotes that it is NOT
	; an IRC bot, but rather a needed config file. (It must be named ~Network.)
	; 

	[~Network]
	; These keys are needed for the bot to operate.
	name = FFSNetwork
	host = irc.ffsnetwork.com
	port = 6667
	owners = westie-cat.co.uk
	; These keys are optional, but are useful to have.
	bind = 193.238.85.98
	channels = #westie, #channel2
	plugins = Evaluation, AutoInvite
	rotation = SEND_DIST
	quitmsg = A quit messag- wait, why are you reading this?
	delimiter = "!"

	[OUTRAGEbot]
	; 
	; This header doesn't have a '~' suffix, which denotes that it
	; is an IRC bot. All three are needed.
	; 
	altnick = OUTRAGEbot`
	username = testing
	realname = David Weston


	; 
	; This is an example of configuration for plugins. What you can do here
	; is use the name of the plugin, with '~' prefixed to it, like '~Network'.
	; You can then call this from PHP with $this->getConfig(); - which will
	; the configuration (for that plugin only) in an array.
	; 
	[~Evaluation]
	; As you can tell, this is for the plugin 'Evaluation'.
	testing = "It seems to work!"
